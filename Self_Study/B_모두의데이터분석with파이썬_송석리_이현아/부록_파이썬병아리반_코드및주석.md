# 2. 연산자 사용하기

```python
print(3 * 10) # 3과 10을 곱한 값 출력 
print(3 ** 10) # 3의 10승 값 출력 
print(3 % 10) # 3을 10으로 나눈 나머지 출력 
print(3 / 10) # 3을 10으로 나눈 값 출력 
print(3 // 10) # 3을 10으로 나눈 몫 값 출력

print(10 >= 3) # 10이 3보다 크거나 같다 > True 출력 
print(10 <= 3) # 10이 3보다 작거나 같다 > False 출력 
print(10 == 3) # 10이 3과 같다 > False 출력 
print(10 != 3) # 10이 3과 같지 않다 > True 출력 
print(3 % 2 == 1) # 3을 2로 나눈 나머지가 1과 같다 > True 출력

print(1 == 1 and 2 != 1) # ‘1과 1은 같다’와 ’2와 1은 같지 않다’ 모두 True이므로 True 출력 / and는 둘 중 하나 이상이 False면 결과가 False, 둘 다 True인 경우만 True 
print(10 % 2 != 0 and 1 + 1 > 0) # ’10을 2로 나눈 나머지가 0과 같지 않다’는 False, ’1과 1을 더한 값은 0보다 크다’는 True이므로 False 출력

print(10 < 5 or 10 == 5) # ’10은 5보다 작다’는 False, ’10은 5와 같다’는 False이므로 False 출력 / or은 둘 중 하나 이상이 True면 결과가 True, 둘다 False인 경우만 False 
print(10 % 2 != 0 or 1 + 1 > 0) # ’10을 2로 나눈 나머지가 0과 같지 않다’는 False, ’1과 1을 더한 값은 0보다 크다’는 True이므로 True 출력

print(not 10 > 5) # ’10이 5보다 크다’는 True이므로 False 출력 / not을 붙이면 반대로 
print(not 10 == 5) # ’10과 5는 같다’는 False이므로 True 출력 
print(not 0) # 숫자 0은 False이므로 True 출력 
print(not 10) # 0이 아닌 숫자 10은 True 이므로 False 출력
```

# 5. 순서 있는 저장 공간, 리스트

```python
names = [‘쵸파’, ‘루피’, ‘상디’, ‘조로’] 
print(names[0]) # names에 지정된 리스트의 1번째 값 출력 
print(names[1]) # names에 지정된 리스트의 2번째 값 출력 
print(names[2]) # names에 지정된 리스트의 3번째 값 출력 
print(names[3]) # names에 지정된 리스트의 4번째 값 출력

print(names[-1]) # names에 지정된 리스트의 마지막 값 출력
print(names[0:2]) # names에 지정된 리스트의 1번째부터 2번째 값까지 출력 
print(names[1:3]) # names에 지정된 리스트의 2번째부터 3번째 값까지 출력 
print(names[1:]) # names에 지정된 리스트의 1번째부터 마지막 값까지 출력 
print(names[:]) # names에 지정된 리스트의 모든 값 출력

names.append(‘나미’) # 기존 names에 지정된 리스트 마지막에 ‘나미’ 추가 
print(names) # ’나미’가 추가된 리스트가 출력

print(len(names)) # names에 지정된 리스트의 길이(요소 개수) 출력 
print(len(‘data analysis for everyone’)) # 문자열의 길이 출력

names = [‘쵸파’, ‘루피’, ‘상디’, ‘조로’] # 변수에 리스트 지정 
names.append(‘해적왕’) # 지정한 리스트의 마지막에 ‘해적왕’ 추가 
for name in names # names에 존재하는 요소들에 대해 차례대로 아래 내용 수행 
	if len(name) > 2: # 요소의 문자 길이가 2보다 크면, 
		print(name, ‘왔나요~?’) # ‘(name) 왔나요~?’ 형태로 출력
```
